SELECT (
SELECT tt.nro_com FROM trap.tram tt where tt.ci_per='8621080' and tt.id_dia = '9130' LIMIT 1)
AS nro_com, 
TRIM(vt.cod_val) AS cod_val,
(
SELECT vv.des_val FROM val.valores vv WHERE vv.cod_val = vt.cod_val )
AS des_val,
vt.imp_val, trim(vt.ci_per) as ci_per, des_per, vt.fec_tra as fec_tra
FROM val.tra_dia vt
WHERE vt.id_dia = $P{p_id_dia}
  AND vt.ci_per = $P{p_ci_per}
  AND vt.usr_cre = $P{p_usr_cre}
  AND vt.gestion = $P{p_gestion}
